16. prakīrṇaka vagga

Brough 259' data-uid='gdhp259ekasaṇa ekasaya
ekaïyaï savudu
eku ramahi atvaṇa
arañi ekao vasa ◦

Brough 260' data-uid='gdhp260yasa ṣadha i praña ya
viya otrapia hiri
so ho mahadhaṇa bhodi
moham aña baho dhaṇa ◦

Brough 261' data-uid='gdhp261na sadi putra traṇaï
na bhoa na vi banava
adeṇa abhiduṇasa
nasti ñadihi traṇadha ◦

Brough 262' data-uid='gdhp262drupravaï druabhiramu
druaj̄avasaṇa ghara
dukhu samaṇasavaso
dukhaṇuvadida bhava ◦

Brough 263' data-uid='gdhp263kiche maṇuśapradilabhu
kicha macaṇa jivida
kiche sadhamaśramaṇa
kiche budhaṇa upaya ◦

Brough 264' data-uid='gdhp264sukaraṇi asadhuṇi
atvaṇo ahidaṇa yi
ya du hida ji sadhu ji
ta gu paramadrukara ◦

Brough 265' data-uid='gdhp265apaṇatha paratheṇa
na kuda yiṇo havaï
atvatha paramu ñatva
svakathaparamu sia

Brough 266' data-uid='gdhp266ayoi yuji atvaṇa
yoaseva ayujadu
atha hitva priagaha
svihadi arthaṇupaśiṇo

Brough 267' data-uid='gdhp267nedi hi mahavira
sadhameṇa tadhakada
dhameṇa neamaṇaṇa
ka y asua viaṇadu ◦

Brough 268' data-uid='gdhp268alia bhaṣamaṇasa
avakamadi devada
muha ji pudia bhodi
sagaṭ́haṇa i bhatsadi ◦

Brough 269' data-uid='gdhp269abhudavadi naraka uvedi
yo yavi kitva na karodi āha
uvhaï ami preca sama bhavadi
nihiṇakama maṇuya paratri

Brough 270' data-uid='gdhp270catvari ṭ́haṇaṇi naro pramatu
avajadi paradarovasevi
amuñalabha aniamasaya
nina tridia niraya caüṭ́ha

Brough 271' data-uid='gdhp271na pareṣa vilomaṇi
na pareṣa kidakida
atvaṇo i samikṣea
samaṇi viṣamaṇi ca ◦

Brough 272' data-uid='gdhp272supaśi vaja añeṣa
atvaṇo maṇa drudaśa
pareṣa eṣu vajaṇa
upuṇadi yatha busu
atvaṇo maṇa chadedi
kali va kidava śaḍha ◦

Brough 273' data-uid='gdhp273alajidavi lajadi

abhayi bhayadarśavi
bhayi abhayadarśaṇo
michadiṭhisamadaṇa
satva gachadi drugadi
