14. paṇḍita vagga

Brough 224' data-uid='gdhp224dhamapridi suhu śayadi
viprasaneṇa cedaso
ariapravedidi dharmi
sada ramadi paṇidu ◦

Brough 225' data-uid='gdhp225yatha vi rada gammiro
viprasano aṇavilo
emu dhamu ṣuṇitvaṇa
viprasidadi paṇida ◦

Brough 226' data-uid='gdhp226sarvatra ya sapuruṣa vivedi
na kamakama lavayadi dhira
suheṇa phuṭha adha va duheṇa
na ucavaya paṇida daśayadi

Brough 227' data-uid='gdhp227atmaṇam eva pradhamu
pradiruvi niveśaï
tadañi aṇuśaśea
na kiliśea paṇidu ◦

Brough 228' data-uid='gdhp228aṣadhehi kradavehi
phiśuṇehi vivhudananahi
sakha na karia paṇido
sag̱adi kavuruṣehi paviya

Brough 229' data-uid='gdhp229ṣadhehi ya peśalehi ya
śilavada yi bahoṣudehi ya
sakha kuvia paṇido
sagadi sapuruṣehi bhadia ◦

Brough 230' data-uid='gdhp230anuśaśadi ovadadi
asabhe hi navaraï
paṇidaṇa prio bhodi
balaṇa bhodi aprio ◦

Brough 231' data-uid='gdhp231nisedara pravatara
yo paśi vajidaśaṇa
nigiśavadi medhavi
tadiśa paṇada bhayi ◦
tadi bhayamaṇaṇa
ṣeho bhodi na paviu

Brough 232' data-uid='gdhp232kaeṇa kuśala kitva
vayaï kuśala baho
maṇeṇa kuśala kitva
apramaṇa nirovadhi
avavaji suhi loke
paṇidu uvavajadi ◦

Brough 233' data-uid='gdhp233yavajiva bi ya balu
paṇida payuvasadi
neva dhamu viaṇadi
praña hisa na vijadi

Brough 234' data-uid='gdhp234muhuta bi ya viñu
paṇada payuvasadi
so du dhamu viaṇadi
praña hisa tadovia

Brough 235' data-uid='gdhp235na abhaṣamaṇa jaṇadi
miṣa balehi paṇida
bhaṣamaṇa du jaṇadi
deśada amuda duduhi ◦

Brough 236' data-uid='gdhp236deśehi jodaï dhamu
uṣivi iṣiṇa dhayu
subhaṣidadhaya hi iṣayu
dhamo ha iṣiṇa dhayu

Brough 237' data-uid='gdhp237poraṇam ida adura
na ida ajetaṇa iva
ninadi tuṣ̄ibhaveṇa
ninadi bahobhaṇiṇo
manabhaṇi vi ninadi
nasti loki aninia

Brough 238' data-uid='gdhp238praśaj̄a śaśvada nasti
nina nica na vijadi
tasa ninapraśaj̄aṣu
na sammijadi paṇida ◦

Brough 239' data-uid='gdhp239śelu yadha ekakhaṇo
vadeṇa na sabhijadi
emu ninapraśaj̄aṣu
na sammijadi paṇida ◦

Brough 240' data-uid='gdhp240ekada ninido prodhu
ekada ji praśaj̄idu
na i aha na i bheṣida
na yi edarahi vijadi ◦

Brough 241' data-uid='gdhp241yo nu ho viña praśaj̄adi
aṇuija śuhaśuhu
achidravuti medhavi
prañaśilasamahida

Brough 242' data-uid='gdhp242nikhu jabodaṇaseva
ko ṇa ninidu arahadi
deva mi ṇa praśajadi
bramoṇa vi praśajidu
